  • Sep 25, 2017
  • One minute read
  1. 判断机型（Ios/Android）
function getBrowserVersion() {
	var ua = navigator.userAgent || navigator.vendor || window.opera;
	var uaInfo = {
	ios: !!ua.match(/\(i[^;]+;( U;)? CPU.+Mac OS X/i),
	android: ua.match(/Android/i) || ua.match(/Linux/i)
	};
	var browserVersion = {
	isIos: function () {
	return uaInfo.ios;
	},
	isAndroid: function () {
	return uaInfo.android;
	}
	};
	return browserVersion;
}

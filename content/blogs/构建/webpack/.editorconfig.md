# 对所有文件生效
[*]
charset=utf-8
end_of_line=crlf
indent_style=space
indent_size=4
insert_final_newline=true		// 末尾添新行
trim_trailing_whitespace=true	// 末尾有空格提示

# 对.scss文件单独设置
[*.scss]
indent_style=space
indent_size=4

# 对.md文件单独设置
[*.md]
trim_trailing_whitespace=false
